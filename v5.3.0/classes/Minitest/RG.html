def puts output = nil
  return io.puts if output.nil?

  if output =~ /(\d+) failures, (\d+) errors/
    if Regexp.last_match[1] != "0" || Regexp.last_match[2] != "0"
      io.puts "\e[31m#{output}\e[0m"
    else
      io.puts "\e[32m#{output}\e[0m"
    end
  else
    io.puts output
  end
end
